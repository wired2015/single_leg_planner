1%GENERATETRMATRICES Generates each of the homogeneous transformation
 2%matrices that describe the kinematic chain between the Sherpa_TT rover's
 3%body coordinate frame and its wheel contact frame. Denavit-Hartenburg
 4%parameters are used to express the transformation between each coordinate
 5%frame in the kinematic chain.
 6%
 7%Inputs:
 8%-uG: A 1x3 vector giving the [x y z] relationship between the body and
 9%coordinate frame
10%-q: A 1x4 vector describing the leg's joint state. This vector includes 
11%[alpha beta gamma].
12%-kC: A struct containing the kinematic parameters of the Sherpa_TT leg.
13%-legNum: The number of the leg that is being considered (1,2,3 or 4).
14%
15%Outputs:
16%TB2G: Transformation from the body to the ground.
17%TP2B: Transformation from the pan joint to the body.
18%TI2P: Transformation from the inner leg joint to the pan joint.
19%TJ2I: Transformation from the inner leg knee joint to the inner leg joint.
20%TO2J: Transformation from the outer leg joint to the inner leg knee joint.
21%TQ2O: Transformation from the outer leg end joint to the outer leg joint.
22%TR2Q: Transformation from the steering base joint to the outer leg end
23%joint.
24%TS2R: Transformation from the steering joint to the steering base joint.
25%TW2S: Transformation from the wheel joint to the steering joint.
26%TC2W: Transformation from the wheel contact point to the wheel joint.
27%
28%generateTrMatrices.m
29%author:    wreid
30%date:      20140214
31
32function [TB2G,TP2B,TI2P,TJ2I,TO2J,TQ2O,TR2Q,TS2R,TW2S,TC2W] = generateTrMatrices(uG,q,kC,legNum)    
33
34    alpha = q(1);
35    beta = q(2);
36    gamma = q(3);
37    phi = q(4);
38        
39    %TODO: Use a 6-DOF relationship between the ground and body frames by
40    %including the roll, pitch and yaw of the platform.
41    TB2G = [eye(3) [uG(1); uG(2); uG(3)]; 0 0 0 1];
42    TP2B = trDH(kC.legAngleOffset(legNum),kC.B2PZOffset,kC.B2PXOffset,0);
43    TI2P = trDH(alpha,kC.l1,kC.l2,-pi/2);
44    TJ2I = trDH(beta,0,kC.l3,0);
45    TO2J = trDH(-beta+kC.zeta,0,kC.l4,0);
46    TQ2O = trDH(gamma,0,kC.l5,0);
47    TR2Q = trDH(-gamma-kC.zeta,0,-kC.l7,-pi/2);
48    TS2R = trDH(0,kC.l6,0,0);
49    TW2S = trDH(phi+pi/2,kC.l8,0,pi/2);
50    TC2W = trDH(0,0,0,pi/2)*trDH(pi/2,-kC.r,0,0);
51
52end
