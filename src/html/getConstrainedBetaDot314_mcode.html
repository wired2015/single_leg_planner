1%GETCONSTRAINEDBETAADOT This function calculates the velocity of
 2%gamma given a pan height constraint and an independent beta angle.
 3%
 4%getConstrainedBetaDot.m
 5%author: wreid
 6%date: 20150224
 7
 8function betaDot = getConstrainedBetaDot(kC,qDot,q)
 9
10    gammaDot = qDot(3);
11    beta = q(2);
12    gamma = q(3);
13
14    betaDot = (-gammaDot*kC.l5*cos(kC.zeta+gamma))/(kC.l3*cos(beta));
15
16end
