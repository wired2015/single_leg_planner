1%LININTERP Returns a Cartesian path that is linearly interpolated between
 2%two points.
 3%
 4%Inputs:
 5%s1: The initial state that is represented by a 1x8 vector. The vector has
 6%the form [t x y z xDot yDot zDot interpBoolean]
 7%s2: The goal state that is represented by a 1x8 vector. The vector has
 8%the form [t x y z xDot yDot zDot interpBoolean]
 9%N: The number of waypoints to be placed in the path.
10%
11%Outputs:
12%path: A 8xN matrix that returns a path from s2 to s1.
13%
14%linInterp.m
15%author: wreid
16%date: 20150224
17
18function path = linInterp(s1,s2,N)
19    
20    tStart = s1(1);
21    cost2Go = s1(2);
22    n1 = s1(3:5);
23    n2 = s2(3:5);
24
25    oldState = n1;
26    path = zeros(N,9);
27    for i = 1:N
28        newState = (n1+i/N*(n2-n1));
29        cost2Go = cost2Go+norm(newState-oldState);
30        path(i,:) = [tStart+i/N cost2Go newState 0 0 0 true]; 
31        oldState = newState;
32    end
33    
34end
