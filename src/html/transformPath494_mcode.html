 99function pathC = transformPath(pathJ,kC,TP2B)
100    [pathH,~] = size(pathJ);
101    pathC = zeros(pathH,9);
102    dist2Go = 0;
103    for i = 1:pathH
104        uP = sherpaTTFK(pathJ(i,2:4),kC);
105        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
106        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
107        uBDot = TP2B(1:3,1:3)*uPDot;
108        if i ~= 1
109            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
110        end
111        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
112    end
113end
