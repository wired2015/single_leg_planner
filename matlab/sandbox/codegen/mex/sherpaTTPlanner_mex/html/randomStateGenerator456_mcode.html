1%RANDOMSTATEGENERATOR Generates N number of random states that are valid in
 2%the Sherpa_TT leg's configuration space.
 3%
 4%Inputs:
 5%-N: The number of random states to be generated.
 6%-jointLimits: A 2x10 array of joint angular limits and angular positions.
 7%-kC: A structure containing the kinematic constants of the Sherpa TT leg.
 8%-panHeight: The height of the pan coordinate frame relative to the the
 9%wheel contact point.
10%-legNum: The leg identifier.
11%
12%Outputs:
13%-states: A Nx6 array. Each row of the array describes a single state. A
14%state has the structure [x y z xDot yDot zDot]. The state coordinates are
15%realtive to the body frame.
16%
17%randomStateGenerator.m
18%author: wreid
19%date: 20150305
20
21function states = randomStateGenerator(N,jointLimits,kC,panHeight,legNum)
22
23    legNum = legNum + 1;
24    if legNum > 4
25        legNum = int32(4);
26    elseif legNum < 1
27        legNum = int32(1);
28    end
29
30    states = zeros(N,6);
31    
32    cartesianLimits = zeros(1,4);
33    betaMin = jointLimits(1,2);
34    betaMax = jointLimits(2,2);
35    gammaMin = jointLimits(1,3);
36    gammaMax = jointLimits(2,3);
37    u1 = sherpaTTFK([0 betaMin gammaMin],kC);
38    u2 = sherpaTTFK([0 betaMax gammaMax],kC);
39    u3 = sherpaTTFK([0 betaMin gammaMax],kC);
40    u4 = sherpaTTFK([0 betaMax gammaMin],kC);
41    cartesianLimits(1) = u1(3);
42    cartesianLimits(2) = u2(3);
43    cartesianLimits(3) = u3(3);
44    cartesianLimits(4) = u4(3);
45
46    for i = 1:N
47        states(i,:) = [randomPoint(jointLimits,cartesianLimits,panHeight,kC,legNum) 0 0 0];
48    end
49    
50    %save('experimentStates','states')
51
52end
