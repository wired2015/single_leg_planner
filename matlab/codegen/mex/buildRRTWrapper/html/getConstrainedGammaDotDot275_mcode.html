1%GETCONSTRAINEDGAMMADOTDOT This function calculates the acceleration of
 2%gamma given a pan height constraint and an independent beta angle.
 3
 4function gammaDotDot = getConstrainedGammaDotDot(kC,qDotDot,qDot,q)
 5
 6    betaDotDot = qDotDot(2);
 7    betaDot = qDot(2);
 8    gammaDot = qDot(3);
 9    beta = q(2);
10    gamma = q(3);
11
12    gammaDotDot = (-betaDotDot*kC.l3*cos(beta)+betaDot^2*kC.l3*sin(beta)+gammaDot^2*kC.l5*sin(kC.zeta+gamma))/(kC.l5*cos(kC.zeta+gamma));
13
14end
