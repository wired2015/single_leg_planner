1%GETCONSTRAINEDBETADOTDOT This function calculates the acceleration of
 2%beta given a pan height constraint and an indpendent gamma angle.
 3
 4function betaDotDot = getConstrainedBetaDotDot(kC,qDotDot,qDot,q)
 5
 6    gammaDotDot = qDotDot(3);
 7    betaDot = qDot(2);
 8    gammaDot = qDot(3);
 9    beta = q(2);
10    gamma = q(3);
11
12    betaDotDot = (-gammaDotDot*kC.l5*cos(kC.zeta+gamma) + gammaDot^2*kC.l5*sin(kC.zeta+gamma) - betaDot^2*kC.l3*sin(beta))/(kC.l3*cos(beta));
13end
