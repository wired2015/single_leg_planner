1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-kC:
 11%-legNum:
 12%-uBDot:
 13%
 14%Outputs:
 15%-T:
 16%-pathC:
 17%-pathJ:
 18%-success:
 19%
 20%buildRRTWrapper.m
 21%author: wreid
 22%date: 20150502
 23
 24function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,kC,legNum,uBDot)
 25
 26    NUM_NODES = int32(1500);
 27    NODE_SIZE = int32(13);
 28    U_SIZE = int32(5);
 29    ankleThreshold = 5*pi/180;
 30    exhaustive = false;
 31    goalSeedFreq = int32(20);
 32    cartesianLimits = [-0.2930   -1.1326   -0.6710   -0.7546];
 33    dt = 0.1;
 34    Dt = 0.7;
 35    HGAINS = [1 0 0.5];
 36    threshold = 0.005;
 37    stepAccRatio = 14;
 38    eta = Dt/stepAccRatio;
 39    U = eta*[1 0; -1 0; 0 1; 0 -1; 0 0];     
 40    bodyHeight = -nInitCartesianB(3);
 41    panHeight  = getPanHeight(bodyHeight,kC);
 42    
 43
 44    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 45    %to the pan coordinate frame.
 46    TP2B = trP2B(kC,legNum);
 47    TB2P = trInv(TP2B); %inv(TP2B);%
 48    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 49    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 50    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 51    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 52    
 53    %Transform the Cartesian goal and final positions in the pan coordinate
 54    %frame to the joint space.
 55    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 56    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 57    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 58    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 59    
 60    nInitJoint = [qInit phiInit 0 qDotInit' 0 omegaInit];
 61    nGoalJoint = [qGoal 0 0 qDotGoal' 0 0];
 62    
 63    %Check that the initial and final positions are valid. If they are not
 64    %return failure and an empty path.
 65    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 66        success = true;
 67        %Run buildRRT.
 68        nInit = [1 0 0 nInitJoint];
 69        nGoal = [0 0 0 nGoalJoint];
 70        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
 71        %Transform path back to the Cartesian space.
 72        pathC = transformPath(pathJ,kC,TP2B);
 73    else
 74        success = false;
 75        pathC = [];
 76        pathJ = [];
 77        T = [];
 78    end
 79    
 80    %Linearly interpolate to the goal state from the final state.
 81    if isempty(pathC)
 82        sFinalC = [0 0 nInitCartesianB true];
 83    else
 84        sFinalC = pathC(end,:);
 85        sGoalC = [0 0 nGoalCartesianB true];
 86        pathCorrection = linInterp(sFinalC,sGoalC,10);
 87        pathC = [pathC; pathCorrection];
 88        [h,~] = size(pathCorrection);
 89        for i = 1:h
 90            uB = TB2P(1:3,1:3)*pathCorrection(i,3:5)' + TB2P(1:3,4);
 91            q = sherpaTTIK(uB',kC,jointLimits);
 92            pathJ = [pathJ; [pathCorrection(i,1) q 0 0 0 0 0 0 0]];  
 93        end
 94    end
 95    
 96
 97end
