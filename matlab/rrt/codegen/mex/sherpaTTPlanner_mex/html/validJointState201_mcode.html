1%VALIDJOINTSTATE Checks if a leg's joint state complies with the leg's angular
 2%position and rate limits.
 3%
 4%Inputs:
 5%-state:       A 1x6 vector containing the joint state to be checked.
 6%-jointLimits: A 2x6 vector containing the angular position and rate
 7%              limits.
 8%Outputs:
 9%-valid:       A logical value indicating whether or not state is valid.
10%
11%validJointState.m
12%author: wreid
13%date:   20151402
14
15function valid = validJointState(state,jointLimits)
16    
17    if state(1) < jointLimits(1,1) || state(1) > jointLimits(2,1) ||...
18       state(2) < jointLimits(1,2) || state(2) > jointLimits(2,2) ||...
19       state(3) < jointLimits(1,3) || state(3) > jointLimits(2,3) ||...
20       state(6) < jointLimits(1,6) || state(4) > jointLimits(2,6) ||...
21       state(7) < jointLimits(1,7) || state(5) > jointLimits(2,7) ||...
22       state(8) < jointLimits(1,8) || state(6) > jointLimits(2,8)
23    
24        valid = false;
25    else
26        valid = true;
27    end
28end
