130function pathC = transformPath(pathJ,kC,TP2B)
131    [pathH,~] = size(pathJ);
132    pathC = zeros(pathH,9);
133    dist2Go = 0;
134    for i = 1:pathH
135        uP = sherpaTTFK(pathJ(i,2:4),kC);
136        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
137        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
138        uBDot = TP2B(1:3,1:3)*uPDot;
139        if i ~= 1
140            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
141        end
142        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
143    end
144end
