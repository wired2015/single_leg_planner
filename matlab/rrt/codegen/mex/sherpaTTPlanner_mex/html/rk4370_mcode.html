  5function [xNewFull,transitionArray] = rk4(uIn,uBDot,dt,Dt,xInit,jointLimits,kC,legNum)
  6%rk4 Summary of this function goes here
  7%   Detailed explanation goes here
  8
  9    u = uIn;
 10
 11    numIterations = round(Dt/dt);
 12    xInit = xInit(4:end);
 13    xNew = zeros(1,length(xInit));
 14    xInitOrig = xInit;
 15    transitionArray = zeros(1,(numIterations+1)*10);
 16    transitionArray(1:10) = xInit;
 17        
 18    for i = 1:numIterations
 19
 20        xNew = numIntegrate(xInit,u,kC,dt);
 21        
 22        phiInit = xInit(4);
 23        
 24        alpha = xNew(1);
 25        beta = xNew(2);
 26        gamma = xNew(3);
 27        epsilon = xNew(5);
 28
 29        alphaDot = xNew(6);
 30        betaDot = xNew(7);
 31        gammaDot = xNew(8);
 32        phiDot = xNew(9);
 33        omega = xNew(10);
 34        
 35        alphaDotDot = u(1);
 36        betaDotDot = u(2);
 37        
 38        alphaMax = jointLimits(2,1);
 39        alphaMin = jointLimits(1,1);
 40        betaMax = jointLimits(2,2);
 41        betaMin = jointLimits(1,2);
 42        gammaMax = jointLimits(2,3);
 43        gammaMin = jointLimits(1,3);
 44        phiMax = jointLimits(2,4);
 45        phiMin = jointLimits(1,4);
 46        
 47        alphaDotMax = jointLimits(2,6);
 48        alphaDotMin = jointLimits(1,6);
 49        betaDotMax = jointLimits(2,7);
 50        betaDotMin = jointLimits(1,7);
 51        gammaDotMax = jointLimits(2,8);
 52        gammaDotMin = jointLimits(1,8);
 53        phiDotMax = jointLimits(2,9);
 54        phiDotMin = jointLimits(1,9);
 55        omegaMax = jointLimits(2,10);
 56        omegaMin = jointLimits(1,10);
 57        
 58        %Check pan angular position limits
 59        if alpha > alphaMax || alpha < alphaMin
 60            alpha = xInit(1);
 61            alphaDot = 0;
 62            alphaDotDot = 0;
 63        end
 64        
 65        %Check inner and outer leg angular position limits
 66        if beta > betaMax || beta < betaMin || gamma > gammaMax || gamma < gammaMin
 67            beta = xInit(2);
 68            gamma = xInit(3);
 69            betaDot = 0;
 70            gammaDot = 0;
 71            betaDotDot = 0;
 72        end
 73        
 74        %Check pan angular velocity limits
 75        if alphaDot > alphaDotMax || alphaDot < alphaDotMin
 76            alphaDot = xInit(6);
 77            alphaDotDot = 0;
 78        %else
 79        %    alphaDotDot = uIn(1);
 80        end
 81        
 82        %Check the inner leg velocity limit.
 83        if betaDot > betaDotMax || betaDot < betaDotMin
 84            betaDot = xInit(7);
 85            betaDotDot = 0;
 86            gammaDot = getConstrainedGammaDot(kC,[alphaDot betaDot gammaDot],[alpha beta gamma]);
 87            gammaDotDot = getConstrainedGammaDotDot(kC,[alphaDotDot betaDotDot 0],[alphaDot betaDot gammaDot],[alpha beta gamma]);
 88        end
 89        
 90        %Check the outer leg velocity limit.
 91        if gammaDot > gammaDotMax || gammaDot < gammaDotMin
 92           gammaDot = xInit(8);
 93           gammaDotDot = 0;
 94           betaDot = getConstrainedBetaDot(kC,[alphaDot betaDot gammaDot],[alpha beta gamma]);
 95           betaDotDot = getConstrainedBetaDotDot(kC,[alphaDotDot 0 gammaDotDot],[alphaDot betaDot gammaDot],[alpha beta gamma]);
 96           if betaDot > betaDotMax || betaDot < betaDotMin
 97                betaDot = xInit(7);
 98                betaDotDot = 0;
 99           end
100        end
101        
102%         if betaDot > betaDotMax || betaDot < betaDotMin || gammaDot > gammaDotMax || gammaDot < gammaDotMin
103%             betaDot = xInit(7);
104%             gammaDot = xInit(8);
105%             betaDotDot = 0;
106%         else
107%            betaDotDot = uIn(2);
108%         end
109        
110        %Check the outer leg velocity limit.
111        
112        [phi,omega] = getPhiAndOmega(uBDot,[alphaDot betaDot gammaDot 0],[alpha beta gamma 0],phiInit,kC,legNum);
113        epsilon = epsilon + dt*omega;
114        
115%         %Check if phi is above threshold, if so then stop the leg and drive
116%         %the steering joint until it is in the correct orientation.
117%         if abs(phi-phiInit)/dt > phiDotMax
118%           if phi > phiInit
119%             phiDot = phiDotMax;
120%           else
121%             phiDot = phiDotMin;  
122%           end
123%           alphaDotDot = 0;
124%           betaDotDot = 0;
125%           alphaDot = 0;
126%           betaDot = 0;
127%           gammaDot = 0;
128%           omega = 0;
129%           alpha = xInitOrig(1);
130%           beta = xInitOrig(2);
131%           gamma = xInitOrig(3);
132%         end
133        
134        u = [alphaDotDot betaDotDot];
135        
136        xNew = [alpha beta gamma phi epsilon alphaDot betaDot gammaDot phiDot omega];
137        
138        xInit = xNew;
139        transitionArray((10*i)+1:(10*(i+1))) = xNew;
140    end
141    xNewFull = [zeros(1,3) xNew];
142     
143end
