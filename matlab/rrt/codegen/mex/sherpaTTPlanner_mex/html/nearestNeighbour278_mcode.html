 5function [xNear,transitionArray,d] = nearestNeighbour(x,T,HGAINS,jointLimits,kC,nodeIDCount,NODE_SIZE)
 6%nearestNeigbour Finds the node in the tree closest to x.
 7%   This function scans each node within the tree and finds the node that
 8%   is closest to the xRand node. The nearest node is returned by the
 9%   function. A distance heuristic is  used
10%   Inputs:
11%       x:  The 1xn state that each node in the tree will be compared to,
12%           to find the node with the minimum distance to it. n refers to
13%           the number of dimensions within the state space.
14%       T:  The nxm tree being searched, m is the number of possible nodes
15%           within the tree.
16%       HGAINS: The gains applied to the heuristic function.
17%   Outputs:
18%       xNear:  The node in the tree that is closet to x.
19
20    %Iterate over the entire tree and apply the distance heuristic function
21    %to each node.
22    d = zeros(1,nodeIDCount);
23    
24    %parfor i = 1:nodeIDCount
25    for i = 1:nodeIDCount
26        d(i) = heuristicSingleLeg(x,T(i,:),HGAINS,jointLimits,kC);
27    end
28    
29    [d,minIndex] = min(d(1:nodeIDCount));
30    %[d,minIndex] = min(d(1:nodeIDCount));
31    xNear = T(minIndex,1:NODE_SIZE);
32    transitionArray = T(minIndex,(NODE_SIZE+1):end);
33end
